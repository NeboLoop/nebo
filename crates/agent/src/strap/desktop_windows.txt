### os — Desktop Automation (resource: window/input/clipboard/...)
USE THIS when: user wants to interact with the GUI — click, type, screenshot, manage windows, use menus, or control desktop apps visually.

- os(resource: "input", action: "click", x: 100, y: 200) — Click at coordinates
- os(resource: "input", action: "click", element: "B3") — Click element from screenshot
- os(resource: "input", action: "double_click", x: 100, y: 200) — Double-click
- os(resource: "input", action: "right_click", x: 100, y: 200) — Right-click
- os(resource: "input", action: "type", text: "hello") — Type text
- os(resource: "input", action: "hotkey", keys: "ctrl+c") — Keyboard shortcut
- os(resource: "input", action: "scroll", direction: "down", amount: 3) — Scroll
- os(resource: "input", action: "move", x: 100, y: 200) — Move cursor
- os(resource: "input", action: "drag", x: 10, y: 10, to_x: 200, to_y: 200) — Drag
- os(resource: "input", action: "paste", text: "content") — Paste via clipboard
- os(resource: "ui", action: "tree", app: "Notepad") — Get UI element hierarchy
- os(resource: "ui", action: "find", app: "Notepad", role: "button", label: "Submit") — Find elements
- os(resource: "ui", action: "click", app: "Notepad", role: "button", label: "Submit") — Click element
- os(resource: "ui", action: "get_value", app: "Notepad", role: "textfield", label: "Search") — Read value
- os(resource: "ui", action: "set_value", app: "Notepad", role: "textfield", label: "Search", value: "query") — Set value
- os(resource: "ui", action: "list_apps") — List apps with UI Automation access
- os(resource: "window", action: "list") — List all windows
- os(resource: "window", action: "focus", app: "Notepad") — Focus window
- os(resource: "window", action: "move", app: "Notepad", x: 0, y: 0) — Move window
- os(resource: "window", action: "resize", app: "Notepad", width: 800, height: 600) — Resize
- os(resource: "window", action: "minimize", app: "Notepad") — Minimize
- os(resource: "window", action: "maximize", app: "Notepad") — Maximize
- os(resource: "window", action: "close", app: "Notepad") — Close window
- os(resource: "shortcut", action: "list") — List available shortcuts
- os(resource: "shortcut", action: "run", name: "My Shortcut") — Run a shortcut
- os(resource: "menu", action: "list", app: "Notepad") — List menu bar items
- os(resource: "menu", action: "menus", app: "Notepad", menu: "File") — List menu items
- os(resource: "menu", action: "click", app: "Notepad", menu: "File", item: "New Window") — Click menu item
- os(resource: "menu", action: "status") — List system tray icons
- os(resource: "menu", action: "click_status", name: "...") — Click a system tray icon
- os(resource: "dialog", action: "detect", app: "Notepad") — Detect open dialogs/modals
- os(resource: "dialog", action: "list", app: "Notepad") — List dialog elements
- os(resource: "dialog", action: "click", app: "Notepad", button_label: "OK") — Click dialog button
- os(resource: "dialog", action: "fill", app: "Notepad", field: "...", value: "...") — Fill a dialog field
- os(resource: "dialog", action: "dismiss", app: "Notepad") — Dismiss dialog
- os(resource: "space", action: "list") — List virtual desktops
- os(resource: "space", action: "switch", direction: "right") — Switch virtual desktop
- os(resource: "space", action: "move_window", direction: "right") — Move window to adjacent desktop
- os(resource: "screenshot", action: "capture") — Capture current screen
- os(resource: "screenshot", action: "capture", format: "file") — Capture and save to disk (returns inline image URL)
- os(resource: "screenshot", action: "see", app: "Notepad") — Capture specific app with annotated element IDs
- os(resource: "tts", action: "speak", text: "Hello") — Text-to-speech
Note: Menu bar automation works best with classic Win32 apps (Notepad, WordPad, Office). UWP and Electron apps may have limited menu access.
